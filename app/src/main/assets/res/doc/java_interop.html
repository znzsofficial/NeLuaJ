Java 互操作

本文汇总 NeLuaJ+ / LuaJ++ 中与 Java 交互的常用方式，内容提取自「便捷 API」与「LuaJ++」文档，便于单独查阅。

1. 绑定类与创建实例

-- 绑定 Java 类
local File = luajava.bindClass "java.io.File"
local f = File("/sdcard/test.txt")

-- 创建实例
local list = luajava.newInstance("java.util.ArrayList")

-- import 语法（LuaJ++）
import "android.widget.TextView"
import "java.util.*", "java.io.*"
import str "java.lang.String"  -- 别名

local tv = TextView(activity)

2. 表与数组转换

-- Java 集合 / 数组转 Lua 表
local luaTable = luajava.astable(javaList)

-- 创建 Java 数组
local a = int[10]
local b = int{1, 2, 3}
local c = Integer[10]

-- 数组长度与下标
print(#a)
print(a[1])      -- 或 a["1"] 视实现而定

3. 类扩展：array / new / class / override

Object.array{}     -- 创建数组
print(Object.new)  -- 构造器
print(Object.class) -- Class 对象
Object.override{}  -- 覆盖方法（见下文）

4. 构造实例与接口实现

-- 直接 () 构造
local b = ArrayList()
local m = HashMap()

-- 接口 / 抽象类表实现
local i = interface {
  methodName = function(arg)
  end
}
local c = abstract {
  methodName = function(super, arg)
  end
}

-- 单方法接口可直接传函数
local r1 = interface(function(arg)
  print(arg)
end)

-- 已绑定的接口类同样支持
Runnable(function()
  print("run")
end)

5. createProxy 代理

-- 字符串接口名 + 实现表
local runnable = luajava.createProxy("java.lang.Runnable", {
  run = function()
    print("running")
  end
})

-- 单方法接口可直接传函数
local runnable2 = luajava.createProxy("java.lang.Runnable", function()
  print("running")
end)

-- 已绑定的 Class
local Runnable = luajava.bindClass("java.lang.Runnable")
local runnable3 = luajava.createProxy(Runnable, function()
  print("running")
end)

-- 多接口：最后一个参数为表或函数；按全部接口查找方法
local proxy = luajava.createProxy(
  Map,
  Closeable,
  {
    get = function(key)
      return key
    end,
    close = function()
      print("closed")
    end
  }
)

未显式实现 equals / hashCode / toString 时，使用 Java 默认对象语义，避免作为 Map 键时匹配失败。

6. override 覆盖方法

override 生成增强子类，只覆盖表中提供的方法。覆盖方法的第一个参数是 superCall，调用可执行父类原方法。

local ArrayList = luajava.bindClass("java.util.ArrayList")

local list = ArrayList.override {
  add = function(superCall, value)
    print("add:", value)
    return superCall(value)
  end
}()

-- 简写：类后直接跟实现表再构造
list = ArrayList {
  add = function(s, a)
  end
}

-- 抽象类 / 普通类
local MyObserver = FileObserver.override {
  onEvent = function(superCall, event, path)
    print(event, path)
  end
}
local observer = MyObserver(path)

未在表中提供的方法保持 Java 原有行为。superCall(...) 会按 Java 签名转换参数。

7. 属性与 getter/setter 语法糖

-- set/get 可写成属性
b.setText("")
-- 等价
b.text = ""

t = b.getText()
-- 等价
t = b.text

-- isXxx 简写
view.isActivated()
-- 可写作
view.Activated

-- 批量设置属性
Button(this) {
  text = "test",
  enabled = false
}

8. 监听器与函数式接口

-- 完整写法
mBtn.setOnClickListener(View.OnClickListener {
  onClick = function(v)
    print(v)
  end
})

-- 可省略接口类型
mBtn.setOnClickListener({
  onClick = function(v)
    print(v)
  end
})

-- 单方法接口可直接传函数
mBtn.setOnClickListener(function(v)
  print(v)
end)

-- 属性式
mBtn.onClick = function(v)
  print(v)
end

-- 非必要接口类型可省略
mViewPager.addOnPageChangeListener {
  onPageSelected = function(_)
  end
}

obj.run(function()
  -- do something
end)

9. 元方法与其它

function Button:print()
  print(self)
end
Button(this):print()

-- Java 方法用 . 调用；(!) 表示无参等惯用法（视版本）
b.add(!)

-- 表键可用关键字字面量
t = {}
t.end = 123

10. 类型判断

if luajava.instanceof(view, ViewGroup) then
  -- ...
end
